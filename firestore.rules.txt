// Firebase Console → Firestore Database → Rules tab mein ye paste karo,
// phir "Publish" dabao. Ye file khud deploy nahi hoti — sirf reference ke liye hai.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{uid} {
      // Root document (membership/subscription status) — sirf apna hi padh sakta hai.
      // Write yahan se allow nahi hai kyunki wo sirf webhook (Admin SDK) se hona chahiye.
      allow read: if request.auth != null && request.auth.uid == uid;

      match /kv/{doc} {
        // Watchlist, portfolio, alerts — har user apna hi padh/likh sakta hai.
        allow read, write: if request.auth != null && request.auth.uid == uid;
      }
    }
  }
}
